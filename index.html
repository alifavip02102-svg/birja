    // Обработчик открытия торговых сделок контрактов

    window.executeTrade = function(direction) {

        const amount = parseFloat(document.getElementById('tradeAmount').value) || 0;

        if (amount <= 0) return showToast("Некорректная сумма ордера", "error");

        if (amount > userBalance) return showToast("Недостаточный объем свободной ликвидности на балансе.", "error");

        

        userBalance -= amount;

        document.getElementById('userBalance').innerText = `$${userBalance.toFixed(2)}`;

        allDealsCount++;

        

        const dealUuid = 'TX-' + Math.floor(100000 + Math.random() * 900000);

        activeDeals.push({

            id: dealUuid,

            asset: activeAsset,

            type: direction,

            investment: amount,

            entryPrice: livePrices[activeAsset],

            timeLeft: 15,

            payoutRate: assetsData[activeAsset].payout,

            digits: assetsData[activeAsset].digits,

            status: 'PROCESSING'

        });

        

        showToast(`Контракт ${dealUuid} успешно размещен на рынке.`, "success");

        updateDealsTable();

    };


    // Обновление таблицы ордеров

    function updateDealsTable() {

        const tbody = document.getElementById('dealsTableBody');

        tbody.innerHTML = '';

        

        if (activeDeals.length === 0) {

            tbody.innerHTML = `<tr><td colspan="8" style="text-align:center; padding:30px; color: var(--text-muted);">Активные рыночные контракты отсутствуют</td></tr>`;

            document.getElementById('statsDealsCount').innerText = `ОБРАБОТКА: 0 | СИСТЕМА: ${allDealsCount}`;

            return;

        }


        let processingCount = 0;

        

        // Рендерим от новых к старым

        for(let i = activeDeals.length - 1; i >= 0; i--) {

            const deal = activeDeals[i];

            if (deal.status === 'PROCESSING') processingCount++;

            

            let statusText = '';

            if (deal.status === 'PROCESSING') statusText = '<span style="color:var(--neon-blue);">В обработке</span>';

            else if (deal.status === 'WIN') statusText = `<span style="color:var(--neon-green);">+$${(deal.investment * (1 + deal.payoutRate)).toFixed(2)}</span>`;

            else statusText = `<span style="color:var(--neon-pink);">-$${deal.investment.toFixed(2)}</span>`;


            const currentOrExitPrice = deal.status === 'PROCESSING' ? livePrices[deal.asset] : deal.exitPrice;

            const vectorText = deal.type === 'CALL' ? '<span class="row-call">ВВЕРХ ▲</span>' : '<span class="row-put">ВНИЗ ▼</span>';


            tbody.innerHTML += `

                <tr>

                    <td style="font-family:'Roboto Mono'; font-size:11px; color:var(--text-muted);">${deal.id}</td>

                    <td><b>${deal.asset}</b></td>

                    <td>${vectorText}</td>

                    <td style="font-family:'Roboto Mono';">$${deal.investment.toFixed(2)}</td>

                    <td style="font-family:'Roboto Mono';">${deal.entryPrice.toFixed(deal.digits)}</td>

                    <td style="font-family:'Roboto Mono';">${currentOrExitPrice.toFixed(deal.digits)}</td>

                    <td>${deal.timeLeft > 0 ? deal.timeLeft + ' сек' : 'Закрыт'}</td>

                    <td>${statusText}</td>

                </tr>`;

        }

        

        document.getElementById('statsDealsCount').innerText = `ОБРАБОТКА: ${processingCount} | СИСТЕМА: ${allDealsCount}`;

    }


    // Процессинг тайм-аутов активных ордеров

    function processActiveDeals() {

        let stateChanged = false;

        

        activeDeals.forEach(deal => {

            if (deal.status === 'PROCESSING') {

                deal.timeLeft--;

                stateChanged = true;

                

                if (deal.timeLeft <= 0) {

                    deal.exitPrice = livePrices[deal.asset];

                    const isCallWin = deal.type === 'CALL' && deal.exitPrice > deal.entryPrice;

                    const isPutWin = deal.type === 'PUT' && deal.exitPrice < deal.entryPrice;

                    

                    if (isCallWin || isPutWin) {

                        deal.status = 'WIN';

                        userBalance += deal.investment * (1 + deal.payoutRate);

                        showToast(`Контракт ${deal.id} закрыт в ПРИБЫЛЬ!`, "success");

                    } else {

                        deal.status = 'LOSS';

                        showToast(`Контракт ${deal.id} закрыт вне денег.`, "error");

                    }

                    document.getElementById('userBalance').innerText = `$${userBalance.toFixed(2)}`;

                }

            }

        });

        

        if (stateChanged) updateDealsTable();

    }
